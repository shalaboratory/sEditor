Quick Start

Include the sEditor script in your HTML file and initialize it on any <div>.

<!-- Container -->
<div id="myEditor"></div>

<!-- Script -->
<script src="js/seditor.pro.min.js"></script>
<script>
  sEditor.run({
    selector: '#myEditor'
  });
</script>

Configuration

Customize sEditor to fit your needs by passing options to the run method.

sEditor.run({
  selector: '#editor',
  plugins: {
    image: true,   // Enable image modal
    link: true,    // Enable link modal
    source: true   // Enable HTML source view
  }
});
